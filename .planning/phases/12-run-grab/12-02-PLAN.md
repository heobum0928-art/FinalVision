```csharp
//260406 hbk -- IMG-03: 시간폴더 선택 → Shot1~5 이미지 일괄 로드 (D-05, D-06)
private void Btn_LoadFolder_Click(object sender, RoutedEventArgs e)
{
    var dlg = new Ookii.Dialogs.Wpf.VistaFolderBrowserDialog {
        Description = "시간 폴더 선택 (Shot1~5 이미지 일괄 로드)",
        UseDescriptionForTitle = true,
        SelectedPath = SystemSetting.Handle.ImageSavePath,
    };
    if (dlg.ShowDialog() != true) return;
    string folder = dlg.SelectedPath;

    var seq = _pSeq[ESequence.Inspection];
    if (seq == null) return;

    int loadedCount = 0;
    for (int i = 0; i < seq.ActionCount; i++) {
        var act = seq[i] as Action_Inspection;   //260406 hbk
        if (act == null) continue;
        var p = act.Param as InspectionParam;
        if (p == null) continue;

        //260406 hbk -- act.Name 기준으로 파일 검색 (Research Pitfall 2: ProcessName 대신 Name 사용)
        string[] files = Directory.GetFiles(folder, act.Name + "_*.bmp");
        if (files.Length == 0)
            files = Directory.GetFiles(folder, act.Name + "_*_capture_*.jpg");
        if (files.Length == 0) continue;   //260406 hbk -- 매핑 실패 시 기존 값 유지

        string best = files[files.Length - 1];  //260406 hbk -- 정렬된 배열의 마지막 = 최신
        Array.Sort(files);
        best = files[files.Length - 1];

        p.SimulImagePath = best;   //260406 hbk
        var mat = OpenCvSharp.Cv2.ImRead(best, OpenCvSharp.ImreadModes.Color);
        p.SetOriginalImage(mat);
        mat?.Dispose();
        loadedCount++;
    }

    //260406 hbk -- 현재 탭의 이미지 갱신
    RefreshImage();
    UpdateResultLabel();

    if (loadedCount == 0)
        CustomMessageBox.Show("알림", "폴더에서 매핑 가능한 Shot 이미지를 찾지 못했습니다.", MessageBoxImage.Information);
}
```
