//260406 hbk -- IMG-04: 이미지 저장 경로의 날짜 폴더 목록 반환 (D-10)
public string[] GetImageDateFolders() {
    if (!Directory.Exists(ImageSavePath)) return new string[0];
    return Directory.GetDirectories(ImageSavePath)
        .Where(d => System.Text.RegularExpressions.Regex.IsMatch(Path.GetFileName(d), @"^\d{8}$"))
        .OrderByDescending(d => d)
        .ToArray();
}
```
- using 추가: `using System.Linq;` (이미 존재하면 생략), `using System.Text.RegularExpressions;` 불필요 시 간단한 길이+숫자 체크로 대체 가능

//260406 hbk -- IMG-04: 날짜 폴더 목록용 뷰모델
public class DateFolderItem : System.ComponentModel.INotifyPropertyChanged {
    public string FullPath { get; set; }
    public string Name { get; set; }
    private bool _isChecked;
    public bool IsChecked {
        get => _isChecked;
        set { _isChecked = value; PropertyChanged?.Invoke(this, new System.ComponentModel.PropertyChangedEventArgs(nameof(IsChecked))); }
    }
    public event System.ComponentModel.PropertyChangedEventHandler PropertyChanged;
}

//260406 hbk -- 폴더 목록 로드
private void LoadDateFolders() {
    var folders = SystemSetting.Handle.GetImageDateFolders();
    var items = new System.Collections.ObjectModel.ObservableCollection<DateFolderItem>();
    foreach (var f in folders) {
        items.Add(new DateFolderItem { FullPath = f, Name = Path.GetFileName(f), IsChecked = false });
    }
    lb_dateFolders.ItemsSource = items;
}

private void Btn_RefreshFolders_Click(object sender, RoutedEventArgs e) {
    LoadDateFolders();
}

//260406 hbk -- D-11, D-12: 선택된 폴더 삭제
private void Btn_DeleteFolders_Click(object sender, RoutedEventArgs e) {
    var items = lb_dateFolders.ItemsSource as System.Collections.ObjectModel.ObservableCollection<DateFolderItem>;
    if (items == null) return;
    var selected = items.Where(x => x.IsChecked).ToList();
    if (selected.Count == 0) {
        CustomMessageBox.Show("알림", "삭제할 폴더를 선택하세요.", MessageBoxImage.Information);
        return;
    }

    //260406 hbk -- D-11: 삭제 전 확인 다이얼로그
    string names = string.Join(", ", selected.Select(x => x.Name));
    var result = CustomMessageBox.ShowConfirmation(
        "폴더 삭제 확인",
        $"선택된 {selected.Count}개 폴더를 삭제합니까?\n({names})\n\n하위 파일 모두 삭제됩니다.",
        MessageBoxButton.YesNo);
    if (result != MessageBoxResult.Yes) return;

    int deleted = 0;
    foreach (var item in selected) {
        try {
            Directory.Delete(item.FullPath, true);   //260406 hbk -- D-12
            deleted++;
        } catch (Exception ex) {
            CustomMessageBox.Show("삭제 실패", $"{item.Name}: {ex.Message}", MessageBoxImage.Error);
        }
    }

    if (deleted > 0) {
        Logging.PrintLog((int)ELogType.Trace, "[IMG] 폴더 {0}개 삭제 완료", deleted);
        LoadDateFolders();   //260406 hbk -- 목록 새로고침
    }
}
```
