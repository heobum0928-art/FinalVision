private void LoadParamsForCurrentTab() {           //260327 hbk
    int idx = tabControl_shot.SelectedIndex;       //260327 hbk
    if (idx < 0 || idx >= 5) return;               //260327 hbk

    // 포커스 해제 — 편집 중인 PropertyGrid 값 커밋 //260327 hbk
    System.Windows.Input.Keyboard.ClearFocus();     //260327 hbk

    InspectionParam param = _params[idx];           //260327 hbk

    // ROI 초기화: 0,0,0,0 → 전체 이미지 크기 (per D-04) //260327 hbk

            DeviceHandler.INSPECTION_CAMERA_WIDTH,
            DeviceHandler.INSPECTION_CAMERA_HEIGHT);  //260327 hbk

    canvas_roi.SetParam(param);                      //260327 hbk — RuntimeResizer 갱신
    canvas_roi.IsEditable = true;                    //260327 hbk
}
