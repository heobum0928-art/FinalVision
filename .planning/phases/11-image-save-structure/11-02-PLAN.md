```csharp
//260403 hbk -- ImageFolderManager-based save: original + annotated pair (D-03, D-04, D-06, D-07)
private void SaveResultImage(Mat image, bool isOK)
{
    if (image == null) return;

    var setting = SystemSetting.Handle;
    if (isOK && !setting.SaveOkImage) return;    //260326 hbk
    if (!isOK && !setting.SaveNgImage) return;   //260326 hbk

    if (string.IsNullOrEmpty(_FolderPath)) return;   //260403 hbk -- guard: no folder path

    try
    {
        // original image -- async save with Clone (D-03)   //260403 hbk
        string origPath = ImageFolderManager.GetSavePath(_FolderPath, Name, isOK);
        Mat imgClone = image.Clone();
        Task.Factory.StartNew((obj) =>
        {
            var mat = obj as Mat;
            try { mat.SaveImage(origPath); }
            catch (Exception ex) { Logging.PrintLog((int)ELogType.Error, string.Format("SaveImage Error: {0}", ex.Message)); }
            finally { mat.Dispose(); }
        }, imgClone);

        // annotated image -- async save with Clone (D-04, D-06)   //260403 hbk
        Mat annotated = _MyParam.LastAnnotatedImage;
        if (annotated != null && !annotated.IsDisposed)   //260403 hbk -- null guard for SIMUL mode
        {
            string annotatedPath = ImageFolderManager.GetAnnotatedSavePath(_FolderPath, Name, isOK);
            Mat annotatedClone = annotated.Clone();
            Task.Factory.StartNew((obj) =>
            {
                var mat = obj as Mat;
                try { mat.SaveImage(annotatedPath); }
                catch (Exception ex) { Logging.PrintLog((int)ELogType.Error, string.Format("SaveAnnotatedImage Error: {0}", ex.Message)); }
                finally { mat.Dispose(); }
            }, annotatedClone);
        }
    }
    catch (Exception ex)
    {
        Logging.PrintLog((int)ELogType.Error, string.Format("SaveImage Error: {0}", ex.Message));
    }
}
```
