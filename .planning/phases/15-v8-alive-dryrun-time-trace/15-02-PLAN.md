1. 기존 `_aliveResponseReceived` 필드 아래에 상수 추가:

```csharp
private const int ALIVE_SEND_INTERVAL_MS = 1000;  //260410 hbk — 1초 주기 송신
private const int ALIVE_TIMEOUT_MS = 3000;         //260410 hbk — 3초 타임아웃
```

2. AliveProcess() 메서드 추가 (ProcessAlive() 아래):
```csharp
private void AliveProcess() {  //260410 hbk — ALIVE 하트비트 백그라운드 스레드
    Stopwatch aliveTimer = new Stopwatch();
    while (!IsTerminated) {
        if (!Server.IsConnected()) {
            Thread.Sleep(500);  //260410 hbk — 연결 없으면 대기
            continue;
        }

        //260410 hbk — V→Client ALIVE 송신
        _aliveResponseReceived = false;
        SendAlivePacket();
        aliveTimer.Restart();

        //260410 hbk — 3초 내 응답 대기
        while (aliveTimer.ElapsedMilliseconds < ALIVE_TIMEOUT_MS && !IsTerminated) {
            if (_aliveResponseReceived) break;
            Thread.Sleep(50);
        }

        if (IsTerminated) break;

        if (!_aliveResponseReceived) {
            //260410 hbk — 타임아웃: 알람 + 기존 소켓 종료 (Client 재접속 대기)
            PerformAliveTimeout();
        }

        //260410 hbk — 다음 송신까지 남은 시간 대기
        int elapsed = (int)aliveTimer.ElapsedMilliseconds;
        int remaining = ALIVE_SEND_INTERVAL_MS - elapsed;
        if (remaining > 0 && !IsTerminated) {
            Thread.Sleep(remaining);
        }
    }
}

private void SendAlivePacket() {  //260410 hbk
    string msg = "$ALIVE:1@";
    Server.SendMessage(0, msg);  //260410 hbk — 첫 번째 Client에 송신 (MAX_CONNECTION_COUNT=1)
    Logging.PrintLog((int)ELogType.TcpConnection, "[TCP][SEND] ALIVE heartbeat");
}

private void PerformAliveTimeout() {  //260410 hbk — 연결 끊김 판정
    Logging.PrintLog((int)ELogType.Error, "[ALIVE] Timeout — no response within {0}ms. Disconnecting client.", ALIVE_TIMEOUT_MS);
    //260410 hbk — 기존 Client 소켓 강제 종료 → TcpServer가 Accept 대기로 복귀
    if (Server.GetConnectedClientCount() > 0) {
        try {
            Server.GetClient(0).Disconnect();
        } catch (Exception ex) {
            Logging.PrintLog((int)ELogType.Error, "[ALIVE] Disconnect error: {0}", ex.Message);
        }
    }
}
```

주의: `Stopwatch`는 이미 `System.Diagnostics` using이 있으므로 추가 불필요. `IsTerminated`는 base partial class(WPF_Example/SystemHandler.cs)에서 `private bool IsTerminated`로 선언되어 있으며 같은 partial class이므로 접근 가능.
